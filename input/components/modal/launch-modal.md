
# Launch-modal
==============
## Props

### Required:
  - children: ReactNode
  - modalId: string
  - title: string

### Optional:
  - as: 'div' | 'span'
  - blurId: string
  - button: boolean
  - callback: VoidValueCallback<boolean>
  - className: string
  - color: Color | Naked
  - inline: boolean
  - launchClassName: string
  - launchText: string
  - render: ReactNode
  - size: string
  - other: unknown
